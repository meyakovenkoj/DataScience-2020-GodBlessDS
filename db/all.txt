//// -- LEVEL 1
//// -- Tables and References

// Creating tables
Table airport {
  code varchar [pk]
  city varchar
  country varchar
}

Table aircraft {
  type varchar [pk]
  name varchar
}

Table flight {
  id int [pk]
  from_id varchar
  to_id varchar
  from_date date
  til_date date
  week varchar
  dep_time timestamp
  arr_time timestamp
  flight_code varchar
  type varchar
  travel_time timestamp
 }
 
Table cur_flight {
  id int [pk]
  fl_date date
  codeshare boolean
  arrival varchar
  departure varchar
  flight_code varchar
  fl_id int [ref: > flight.id]
}

Table profile {
  id int [pk]
  nickname varchar
  sex varchar
  last_name varchar
  first_name varchar
}

Table profile_fl {
  pr_id int [pk]
  fl_id int [pk]
}

Table loyality {
  id int [pk]
  pr_id int
  status varchar
  programm varchar
  number int
}

// xml yaml diagram begin
Table users as U {
  userid int [pk] // auto-increment
  firstname varchar
  lastname varchar
}

Table cards {
  userid int [ref: > U.userid]
  cardnumber varchar
  //cardstype varchar
  bonusprogrammName varchar
  Indexes {
    (cardnumber) [pk]
  }
}

Table Ticket {
  cardnumber varchar [ref: > cards.cardnumber]
  flightCode varchar
  Class char(1)
}

Table Flight {
  cardnumber varchar [ref: > cards.cardnumber]
  //activitiesType varchar
  //actType varchar
  flightCode varchar
  Date varchar
  Departure varchar
  Arrival varchar
}
Ref: "Flight"."Departure" < "airport"."code"

Ref: "Flight"."Arrival" < "airport"."code"
// xml yaml diagram end
