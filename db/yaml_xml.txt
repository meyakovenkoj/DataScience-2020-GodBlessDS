// Creating tables
Table users as U {
  id int [pk, increment] // auto-increment
  firstname varchar
  lastname varchar
}

Table cards {
  cardnumber int
  cardstype varchar
  bonusprogrammName varchar
  owner_id int [ref: > U.id]
  Indexes {
    (cardnumber) [pk]
  }
}

Table Flight as Act {
  cardnumber int [ref: > cards.cardnumber]
  //activitiesType varchar
  //actType varchar
  flightCode varchar
  Date varchar
  Departure varchar
  Arrival varchar
  Fare varchar
  Class char[1]
}
