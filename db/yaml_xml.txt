// Creating tables
Table users as U {
  id int [pk, increment] // auto-increment
  firstname varchar
  lastname varchar
  created_at timestamp
  country_code int
}

Table cards {
  cardnumber int
  cardstype varchar
  bonusprogramm varchar
  owner_id int [ref: > U.id]
  Indexes {
    (cardnumber) [pk]
  }
}

Table activities as Act {
  cardnumber int [ref: > cards.cardnumber]
  activitiesType varchar
  actType varchar
  Plane varchar
  Date varchar
  Departure varchar
  Arrival varchar
  Fare varchar
  Class char[1]
  Indexes {
    (activitiesType) [pk]
  }
}
