//// -- LEVEL 1
//// -- Tables and References

// Creating tables
Table airport {
  code varchar [pk]
  city varchar
  country varchar
}

Table aircraft {
  type varchar [pk]
  name varchar
}

Table flight {
  id int [pk]
  departure varchar
  arrival varchar
  validity_from date
  validity_to date
  days varchar
  dep_time timestamp
  arr_time timestamp
  flight_code varchar
  type varchar
  travel_time timestamp
 }
 
Table json_flight {
  id int [pk]
  fl_date date
  codeshare boolean
  arrival varchar
  departure varchar
  flight_code varchar
}

Table profile {
  id int [pk]
  nickname varchar
  sex varchar
  last_name varchar
  first_name varchar
}

Table profile_fl {
  pr_id int [pk]
  fl_id int [pk]
}

Table loyalty {
  id int [pk]
  pr_id int
  status varchar
  programm varchar
  number int
}

// Creating references
// You can also define relaionship separately
// > many-to-one; < one-to-many; - one-to-one
Ref: flight.arrival > airport.code  
Ref: flight.departure > airport.code
Ref: flight.type > aircraft.type
Ref: loyalty.pr_id > profile.id
Ref: profile_fl.fl_id > json_flight.id
Ref: profile_fl.pr_id > profile.id
Ref: json_flight.arrival > airport.code
Ref: json_flight.departure > airport.code
